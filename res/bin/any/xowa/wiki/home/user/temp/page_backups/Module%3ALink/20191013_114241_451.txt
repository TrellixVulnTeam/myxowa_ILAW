local p = {}
--{{lnki:a}}
--{{lnke:b}}
--{{lnke.db:simplewiki}}

function p.lnki(frame)
  local page_type = frame.args[1];
  local page_name = '';
  local page_text = '';
  if     page_type == 'import.online' then
    page_name = 'Dashboard/Import/Online';
    page_text = 'Import online';
  elseif page_type == 'import.offline' then
    page_name = 'Dashboard/Import/Offline';
    page_text = 'Import offline';
  else
    page_name = 'MISSING';
    page_text = 'Missing: ' .. page_text;
  end
  return '[[' .. page_name .. '|' .. page_text .. ']]';
end

function p.lnke(frame)
  local page_type = frame.args[1];
  local page_name = '';
  local page_text = '';
  if     page_type == 'wm.dumps' then
    page_name = 'http://dumps.wikimedia.org/backup-index.html';
    page_text = 'Wikimedia database dumps';
  end
  return '[' .. page_name .. ' ' .. page_text .. ']';
end
